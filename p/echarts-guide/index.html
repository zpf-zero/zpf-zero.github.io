import { ref, watch, onMounted, onUnmounted } from 'vue'
import * as echarts from 'echarts'
import type { EChartsOption } from 'echarts'

interface Props {
  option: EChartsOption
  width?: string
  height?: string
  theme?: string
}

const props = withDefaults(defineProps<Props>(), {
  width: '100%',
  height: '400px',
  theme: undefined,
})

const emit = defineEmits<{
  (e: 'click', params: any): void
  (e: 'legendselectchanged', params: any): void
}>()

const chartRef = ref<HTMLElement>()
let chart: echarts.ECharts | null = null

// åˆå§‹åŒ–å›¾è¡¨
const initChart = () => {
  if (chartRef.value) {
    chart = echarts.init(chartRef.value, props.theme)
    chart.setOption(props.option)
    bindEvents()
  }
}

// ç»‘å®šäº‹ä»¶
const bindEvents = () => {
  chart?.on('click', (params) => emit('click', params))
  chart?.on('legendselectchanged', (params) => emit('legendselectchanged', params))
}

// ç›‘å¬ option å˜åŒ–
watch(
  () => props.option,
  (newOption) => {
    chart?.setOption(newOption, true)
  },
  { deep: true }
)

// å“åº”å¼è°ƒæ•´
const handleResize = () => {
  chart?.resize()
}

onMounted(() => {
  initChart()
  window.addEventListener('resize', handleResize)
})

onUnmounted(() => {
  chart?.dispose()
  window.removeEventListener('resize', handleResize)
})

// æš´éœ²æ–¹æ³•
defineExpose({
  getInstance: () => chart,
  resize: handleResize,
})
</script>
