func (tw *TimeWheel) Start() {
    tw.ticker = time.NewTicker(tw.interval)
    go tw.run()

// è¿è¡Œä¸»å¾ªç¯
func (tw *TimeWheel) run() {
    for {
        select {
        case <-tw.ticker.C:
            tw.tick()
        case task := <-tw.addCh:
            tw.addTask(task)
        case id := <-tw.removeCh:
            tw.removeTask(id)
        case <-tw.stopCh:
            tw.ticker.Stop()
            return
        }
    }
}

// æ—¶é—´æ¨è¿›
func (tw *TimeWheel) tick() {
    tw.mu.Lock()
    defer tw.mu.Unlock()
    
    // è·å–å½“å‰æ§½ä½çš„ä»»åŠ¡åˆ—è¡¨
    bucket := tw.buckets[tw.currentPos]
    
    // æ‰§è¡Œåˆ°æœŸä»»åŠ¡
    for e := bucket.Front(); e != nil; {
        task := e.Value.(*Task)
        
        if task.circle > 0 {
            // è¿˜éœ€è¦ç»§ç»­ç­‰å¾…
            task.circle--
            e = e.Next()
            continue
        }
        
        // æ‰§è¡Œä»»åŠ¡
        next := e.Next()
        go task.Callback()
        
        // ä»é“¾è¡¨å’Œç´¢å¼•ä¸­ç§»é™¤
        bucket.Remove(e)
        delete(tw.taskMap, task.ID)
        e = next
    }
    
    // ç§»åŠ¨æŒ‡é’ˆ
    tw.currentPos = (tw.currentPos + 1) % tw.slots
}

// æ·»åŠ ä»»åŠ¡
func (tw *TimeWheel) addTask(task *Task) {
    tw.mu.Lock()
    defer tw.mu.Unlock()
    
    // è®¡ç®—æ§½ä½å’Œåœˆæ•°
    delay := int(task.Delay / tw.interval)
    circle := delay / tw.slots
    pos := (tw.currentPos + delay) % tw.slots
    
    task.circle = circle
    
    // æ·»åŠ åˆ°æ§½ä½
    element := tw.buckets[pos].PushBack(task)
    
    // è®°å½•ä½ç½®
    tw.taskMap[task.ID] = &TaskLocation{
        slot:    pos,
        element: element,
    }
}

// ç§»é™¤ä»»åŠ¡
func (tw *TimeWheel) removeTask(id string) {
    tw.mu.Lock()
    defer tw.mu.Unlock()
    
    if loc, ok := tw.taskMap[id]; ok {
        tw.buckets[loc.slot].Remove(loc.element)
        delete(tw.taskMap, id)
    }
}

// æ·»åŠ å»¶è¿Ÿä»»åŠ¡ï¼ˆå…¬å¼€æ–¹æ³•ï¼‰
func (tw *TimeWheel) AddTask(id string, delay time.Duration, callback func()) {
    tw.addCh <- &Task{
        ID:       id,
        Delay:    delay,
        Callback: callback,
    }
}

// ç§»é™¤ä»»åŠ¡ï¼ˆå…¬å¼€æ–¹æ³•ï¼‰
func (tw *TimeWheel) RemoveTask(id string) {
    tw.removeCh <- id
}

// åœæ­¢æ—¶é—´è½®
func (tw *TimeWheel) Stop() {
    close(tw.stopCh)
}

package timewheel

import (
    "container/list"
    "sync"
    "time"
)

// å±‚çº§æ—¶é—´è½®
type HierarchicalTimeWheel struct {
    levels      []*TimeWheelLevel
    levelCount  int
    tickMs      int64
    startMs     int64
    currentTime int64
    taskMap     map[string]*HierarchicalTaskLocation
    mu          sync.Mutex
    ticker      *time.Ticker
    stopCh      chan struct{}
}

// å•å±‚æ—¶é—´è½®
type TimeWheelLevel struct {
    slots       []*list.List
    slotCount   int
    tickMs      int64
    currentPos  int
}

// ä»»åŠ¡ä½ç½®
type HierarchicalTaskLocation struct {
    level   int
    slot    int
    element *list.Element
    task    *Task
}

// åˆ›å»ºå±‚çº§æ—¶é—´è½®
func NewHierarchicalTimeWheel(tickMs int64, wheelSize int, levelCount int) *HierarchicalTimeWheel {
    htw := &HierarchicalTimeWheel{
        levels:      make([]*TimeWheelLevel, levelCount),
        levelCount:  levelCount,
        tickMs:      tickMs,
        startMs:     time.Now().UnixMilli(),
        currentTime: 0,
        taskMap:     make(map[string]*HierarchicalTaskLocation),
        stopCh:      make(chan struct{}),
    }
    
    // åˆå§‹åŒ–å„å±‚
    levelTick := tickMs
    for i := 0; i < levelCount; i++ {
        htw.levels[i] = &TimeWheelLevel{
            slots:     make([]*list.List, wheelSize),
            slotCount: wheelSize,
            tickMs:    levelTick,
        }
        for j := 0; j < wheelSize; j++ {
            htw.levels[i].slots[j] = list.New()
        }
        levelTick *= int64(wheelSize)
    }
    
    return htw
}

// æ·»åŠ ä»»åŠ¡
func (htw *HierarchicalTimeWheel) AddTask(id string, delayMs int64, callback func()) {
    htw.mu.Lock()
    defer htw.mu.Unlock()
    
    task := &Task{
        ID:       id,
        Delay:    time.Duration(delayMs) * time.Millisecond,
        Callback: callback,
    }
    
    expireTime := htw.currentTime + delayMs
    htw.addTaskInternal(task, expireTime)
}

func (htw *HierarchicalTimeWheel) addTaskInternal(task *Task, expireTime int64) {
    // è®¡ç®—å»¶è¿Ÿ
    delay := expireTime - htw.currentTime
    
    // æ‰¾åˆ°åˆé€‚çš„å±‚çº§
    for level := 0; level < htw.levelCount; level++ {
        levelWheel := htw.levels[level]
        levelRange := levelWheel.tickMs * int64(levelWheel.slotCount)
        
        if delay < levelRange || level == htw.levelCount-1 {
            // è®¡ç®—æ§½ä½
            slot := int((expireTime / levelWheel.tickMs) % int64(levelWheel.slotCount))
            
            // åˆ›å»ºä»»åŠ¡åŒ…è£…
            taskWrapper := &taskWrapper{
                task:       task,
                expireTime: expireTime,
            }
            
            // æ·»åŠ åˆ°æ§½ä½
            element := levelWheel.slots[slot].PushBack(taskWrapper)
            
            // è®°å½•ä½ç½®
            htw.taskMap[task.ID] = &HierarchicalTaskLocation{
                level:   level,
                slot:    slot,
                element: element,
                task:    task,
            }
            return
        }
    }
}

type taskWrapper struct {
    task       *Task
    expireTime int64
}

// æ—¶é—´æ¨è¿›
func (htw *HierarchicalTimeWheel) advanceTime(targetTime int64) {
    htw.mu.Lock()
    defer htw.mu.Unlock()
    
    for htw.currentTime < targetTime {
        htw.currentTime += htw.tickMs
        htw.processLevel(0)
    }
}

func (htw *HierarchicalTimeWheel) processLevel(level int) {
    if level >= htw.levelCount {
        return
    }
    
    levelWheel := htw.levels[level]
    currentSlot := int((htw.currentTime / levelWheel.tickMs) % int64(levelWheel.slotCount))
    
    // æ£€æŸ¥æ˜¯å¦éœ€è¦æ¨è¿›è¯¥å±‚
    if currentSlot == levelWheel.currentPos {
        return
    }
    
    // å¤„ç†å½“å‰æ§½ä½
    bucket := levelWheel.slots[levelWheel.currentPos]
    for e := bucket.Front(); e != nil; {
        tw := e.Value.(*taskWrapper)
        next := e.Next()
        
        if tw.expireTime <= htw.currentTime {
            // æ‰§è¡Œä»»åŠ¡
            go tw.task.Callback()
            delete(htw.taskMap, tw.task.ID)
        } else if level > 0 {
            // é™çº§åˆ°ä¸‹ä¸€å±‚
            bucket.Remove(e)
            delete(htw.taskMap, tw.task.ID)
            htw.addTaskInternal(tw.task, tw.expireTime)
        }
        
        e = next
    }
    
    levelWheel.currentPos = currentSlot
    
    // æ£€æŸ¥æ˜¯å¦éœ€è¦å¤„ç†ä¸Šå±‚
    if currentSlot == 0 && level < htw.levelCount-1 {
        htw.processLevel(level + 1)
    }
}

// å¯åŠ¨
func (htw *HierarchicalTimeWheel) Start() {
    htw.ticker = time.NewTicker(time.Duration(htw.tickMs) * time.Millisecond)
    
    go func() {
        for {
            select {
            case <-htw.ticker.C:
                htw.advanceTime(time.Now().UnixMilli() - htw.startMs)
            case <-htw.stopCh:
                htw.ticker.Stop()
                return
            }
        }
    }()
}

func (htw *HierarchicalTimeWheel) Stop() {
    close(htw.stopCh)
}

    // é‡ç½®è¶…æ—¶
    cm.timeWheel.RemoveTask(connID)
