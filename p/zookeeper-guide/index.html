type ConfigCenter struct {
    client *Client
    cache  sync.Map
}

func NewConfigCenter(servers []string) (*ConfigCenter, error) {
    client, err := NewClient(servers)
    if err != nil {
        return nil, err
    }
    return &ConfigCenter{client: client}, nil
}

func (cc *ConfigCenter) GetConfig(key string) (string, error) {
    // å…ˆæŸ¥ç¼“å­˜
    if val, ok := cc.cache.Load(key); ok {
        return val.(string), nil
    }
    
    // ä» Zookeeper è·å–
    path := "/config/" + key
    data, _, err := cc.client.Get(path)
    if err != nil {
        return "", err
    }
    
    value := string(data)
    cc.cache.Store(key, value)
    
    // å¯åŠ¨ç›‘å¬
    go cc.watchConfig(key)
    
    return value, nil
}

func (cc *ConfigCenter) watchConfig(key string) {
    path := "/config/" + key
    cc.client.WatchData(path, func(data []byte) {
        cc.cache.Store(key, string(data))
        log.Printf("Config %s updated", key)
    })
}

type ServiceRegistry struct {
    client      *Client
    servicePath string
}

func NewServiceRegistry(servers []string, serviceName string) (*ServiceRegistry, error) {
    client, err := NewClient(servers)
    if err != nil {
        return nil, err
    }
    
    servicePath := "/services/" + serviceName
    
    // ç¡®ä¿æœåŠ¡è·¯å¾„å­˜åœ¨
    exists, _, _ := client.Exists(servicePath)
    if !exists {
        client.Create(servicePath, nil, 0)
    }
    
    return &ServiceRegistry{
        client:      client,
        servicePath: servicePath,
    }, nil
}

// æ³¨å†ŒæœåŠ¡å®ä¾‹
func (sr *ServiceRegistry) Register(instanceID, address string) error {
    path := sr.servicePath + "/" + instanceID
    data := []byte(address)
    
    // åˆ›å»ºä¸´æ—¶èŠ‚ç‚¹
    _, err := sr.client.Create(path, data, zk.FlagEphemeral)
    return err
}

// è·å–æœåŠ¡å®ä¾‹åˆ—è¡¨
func (sr *ServiceRegistry) GetInstances() ([]string, error) {
    children, _, err := sr.client.Children(sr.servicePath)
    if err != nil {
        return nil, err
    }
    
    var addresses []string
    for _, child := range children {
        path := sr.servicePath + "/" + child
        data, _, _ := sr.client.Get(path)
        addresses = append(addresses, string(data))
    }
    
    return addresses, nil
}

// ç›‘å¬æœåŠ¡å˜åŒ–
func (sr *ServiceRegistry) Watch(callback func([]string)) {
    go sr.client.WatchChildren(sr.servicePath, func(children []string) {
        var addresses []string
        for _, child := range children {
            path := sr.servicePath + "/" + child
            data, _, _ := sr.client.Get(path)
            addresses = append(addresses, string(data))
        }
        callback(addresses)
    })
}

type DistributedLock struct {
    client   *Client
    lockPath string
    nodePath string
}

func NewDistributedLock(client *Client, lockName string) *DistributedLock {
    return &DistributedLock{
        client:   client,
        lockPath: "/locks/" + lockName,
    }
}

func (dl *DistributedLock) Lock() error {
    // ç¡®ä¿é”ç›®å½•å­˜åœ¨
    exists, _, _ := dl.client.Exists(dl.lockPath)
    if !exists {
        dl.client.Create(dl.lockPath, nil, 0)
    }
    
    // åˆ›å»ºä¸´æ—¶é¡ºåºèŠ‚ç‚¹
    path, err := dl.client.Create(
        dl.lockPath+"/lock-",
        nil,
        zk.FlagEphemeral|zk.FlagSequence,
    )
    if err != nil {
        return err
    }
    dl.nodePath = path
    
    for {
        // è·å–æ‰€æœ‰å­èŠ‚ç‚¹
        children, _, err := dl.client.Children(dl.lockPath)
        if err != nil {
            return err
        }
        
        sort.Strings(children)
        
        // æ£€æŸ¥æ˜¯å¦æ˜¯æœ€å°èŠ‚ç‚¹
        myNode := path[len(dl.lockPath)+1:]
        if children[0] == myNode {
            return nil // è·å–åˆ°é”
        }
        
        // æ‰¾åˆ°å‰ä¸€ä¸ªèŠ‚ç‚¹å¹¶ç›‘å¬
        var prevNode string
        for i, child := range children {
            if child == myNode {
                prevNode = children[i-1]
                break
            }
        }
        
        // ç›‘å¬å‰ä¸€ä¸ªèŠ‚ç‚¹
        prevPath := dl.lockPath + "/" + prevNode
        exists, _, eventCh, _ := dl.client.conn.ExistsW(prevPath)
        if exists {
            <-eventCh // ç­‰å¾…å‰ä¸€ä¸ªèŠ‚ç‚¹åˆ é™¤
        }
    }
}

func (dl *DistributedLock) Unlock() error {
    if dl.nodePath == "" {
        return nil
    }
    return dl.client.Delete(dl.nodePath, -1)
}

type LeaderElection struct {
    client     *Client
    path       string
    nodePath   string
    isLeader   bool
    onBecomeLeader func()
    onLoseLeader   func()
}

func NewLeaderElection(client *Client, path string) *LeaderElection {
    return &LeaderElection{
        client: client,
        path:   path,
    }
}

func (le *LeaderElection) Start() error {
    // ç¡®ä¿é€‰ä¸¾è·¯å¾„å­˜åœ¨
    exists, _, _ := le.client.Exists(le.path)
    if !exists {
        le.client.Create(le.path, nil, 0)
    }
    
    // åˆ›å»ºä¸´æ—¶é¡ºåºèŠ‚ç‚¹
    path, err := le.client.Create(
        le.path+"/node-",
        nil,
        zk.FlagEphemeral|zk.FlagSequence,
    )
    if err != nil {
        return err
    }
    le.nodePath = path
    
    go le.watchLeadership()
    
    return nil
}

func (le *LeaderElection) watchLeadership() {
    for {
        children, _, err := le.client.Children(le.path)
        if err != nil {
            continue
        }
        
        sort.Strings(children)
        
        myNode := le.nodePath[len(le.path)+1:]
        isLeader := children[0] == myNode
        
        if isLeader && !le.isLeader {
            le.isLeader = true
            if le.onBecomeLeader != nil {
                le.onBecomeLeader()
            }
        } else if !isLeader && le.isLeader {
            le.isLeader = false
            if le.onLoseLeader != nil {
                le.onLoseLeader()
            }
        }
        
        if !isLeader {
            // ç›‘å¬å‰ä¸€ä¸ªèŠ‚ç‚¹
            var prevNode string
            for i, child := range children {
                if child == myNode {
                    prevNode = children[i-1]
                    break
                }
            }
            
            prevPath := le.path + "/" + prevNode
            _, _, eventCh, _ := le.client.conn.ExistsW(prevPath)
            <-eventCh
        }
    }
}

func (le *LeaderElection) IsLeader() bool {
    return le.isLeader
}

func NewClientWithRetry(servers []string, maxRetries int) (*Client, error) {
    var conn *zk.Conn
    var err error
    
    for i := 0; i < maxRetries; i++ {
        conn, _, err = zk.Connect(servers, 5*time.Second)
        if err == nil {
            break
        }
        time.Sleep(time.Second * time.Duration(i+1))
    }
    
    if err != nil {
        return nil, err
    }
    
    return &Client{conn: conn}, nil
}
