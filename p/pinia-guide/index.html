  const user = ref<User | null>(null)
  const loading = ref(false)
  const error = ref<string | null>(null)
  
  // åŒæ­¥ action
  function setUser(newUser: User) {
    user.value = newUser
  }
  
  function clearUser() {
    user.value = null
  }
  
  // å¼‚æ­¥ action
  async function login(username: string, password: string) {
    loading.value = true
    error.value = null

      const response = await fetch('/api/login', {
        method: 'POST',

      throw e
    } finally {
      loading.value = false
    }

    await fetch('/api/logout', { method: 'POST' })
    user.value = null
  }
  
  return {
    user,
    loading,
    error,
    setUser,
    clearUser,
    login,
    logout,
  }

import type { PiniaPluginContext } from 'pinia'

export function loggerPlugin({ store }: PiniaPluginContext) {
  // ç›‘å¬æ‰€æœ‰ state å˜åŒ–
  store.$subscribe((mutation, state) => {
    console.log(`[${mutation.storeId}] ${mutation.type}`)
    console.log('Payload:', mutation.payload)
    console.log('New state:', JSON.parse(JSON.stringify(state)))
  })
  
  // ç›‘å¬æ‰€æœ‰ action è°ƒç”¨
  store.$onAction(({ name, store, args, after, onError }) => {
    const startTime = Date.now()
    
    console.log(`[${store.$id}] Action "${name}" started`)
    
    after((result) => {
      console.log(
        `[${store.$id}] Action "${name}" finished in ${Date.now() - startTime}ms`
      )
    })
    
    onError((error) => {
      console.error(`[${store.$id}] Action "${name}" failed:`, error)
    })
  })
}

// composables/useLoading.ts
export function useLoading() {
  const loading = ref(false)
  const error = ref<Error | null>(null)
  
  async function withLoading<T>(fn: () => Promise<T>): Promise<T> {
    loading.value = true
    error.value = null
    
    try {
      return await fn()
    } catch (e) {
      error.value = e as Error
      throw e
    } finally {
      loading.value = false
    }
  }
  
  return { loading, error, withLoading }
}

// åœ¨ Store ä¸­ä½¿ç”¨

  const { loading, error, withLoading } = useLoading()
  const users = ref<User[]>([])
  
  async function fetchUsers() {
    users.value = await withLoading(() => api.get('/users'))
  }
  
  return { users, loading, error, fetchUsers }
})
